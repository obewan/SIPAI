285 std::string get_timestamp() const {
286 auto now = std::chrono::system_clock::now();
287 auto now_c = std::chrono::system_clock::to_time_t(now);
288 std::tm now_tm{};
289 std::stringstream sst;

291 defined(__WIN32) && !defined(__CYGWIN__)
292 localtime_s(&now_tm, &now_c);

296 sst << "[" << std::put_time(&now_tm, "%F %T") << "] ";
